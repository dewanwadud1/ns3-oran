#ifndef ORAN_REPORT_LTE_UE_CQI_H
#define ORAN_REPORT_LTE_UE_CQI_H

#include "oran-report.h"

#include <string>

namespace ns3
{

/**
 * \ingroup oran
 *
 * Report for a CQI measurement of an LTE UE.
 */
class OranReportLteUeCqi : public OranReport
{
  public:
    /**
     * Get the TypeId of the OranReportLteUeCqi class.
     *
     * \return The TypeId.
     */
    static TypeId GetTypeId();
    /**
     * Constructor of the OranReportLteUeCqi class.
     */
    OranReportLteUeCqi();
    /**
     * Destructor of the OranReportLteUeCqi class.
     */
    ~OranReportLteUeCqi() override;
    /**
     * Get a string representation of this Report.
     *
     * \return A string representation of this Report.
     */
    std::string ToString() const override;
    /**
     * Gets the RNTI.
     *
     * \return The RNTI.
     */
    uint16_t GetRnti() const;
    /**
     * Gets the cell ID.
     *
     * \return The cell ID.
     */
    uint16_t GetCellId() const;
    /**
     * Gets the reported CQI.
     *
     * \return The reported CQI.
     */
    uint8_t GetCqi() const;
    /**
     * Gets the component carrier ID.
     *
     * \return The component carrier ID.
     */
    uint16_t GetComponentCarrierId() const;

  private:
    /// The RNTI.
    uint16_t m_rnti;
    /// The cell ID.
    uint16_t m_cellId;
    /// The CQI value.
    uint8_t m_cqi;
    /// The component carrier ID.
    uint16_t m_componentCarrierId;
};

} // namespace ns3

#endif // ORAN_REPORT_LTE_UE_CQI_H
