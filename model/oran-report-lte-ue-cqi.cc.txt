NS_LOG_COMPONENT_DEFINE("OranReportLteUeCqi");
NS_OBJECT_ENSURE_REGISTERED(OranReportLteUeCqi);

TypeId
OranReportLteUeCqi::GetTypeId()
{
    static TypeId tid = TypeId("ns3::OranReportLteUeCqi")
                            .SetParent<OranReport>()
                            .AddConstructor<OranReportLteUeCqi>()
                            .AddAttribute("Rnti",
                                          "The RNTI.",
                                          UintegerValue(),
                                          MakeUintegerAccessor(&OranReportLteUeCqi::m_rnti),
                                          MakeUintegerChecker<uint16_t>())
                            .AddAttribute("CellId",
                                          "The cell ID.",
                                          UintegerValue(),
                                          MakeUintegerAccessor(&OranReportLteUeCqi::m_cellId),
                                          MakeUintegerChecker<uint16_t>())
                            .AddAttribute("Cqi",
                                          "The CQI value.",
                                          UintegerValue(),
                                          MakeUintegerAccessor(&OranReportLteUeCqi::m_cqi),
                                          MakeUintegerChecker<uint8_t>())
                            .AddAttribute("ComponentCarrierId",
                                          "The component carrier ID.",
                                          UintegerValue(),
                                          MakeUintegerAccessor(&OranReportLteUeCqi::m_componentCarrierId),
                                          MakeUintegerChecker<uint16_t>());
    return tid;
}

OranReportLteUeCqi::OranReportLteUeCqi()
{
    NS_LOG_FUNCTION(this);
}

OranReportLteUeCqi::~OranReportLteUeCqi()
{
    NS_LOG_FUNCTION(this);
}

std::string
OranReportLteUeCqi::ToString() const
{
    NS_LOG_FUNCTION(this);

    std::stringstream ss;
    Time time = GetTime();

    ss << "OranReportLteUeCqi("
       << "E2NodeId=" << GetReporterE2NodeId()
       << ";Time=" << time.As(Time::S)
       << ";RNTI=" << +m_rnti
       << ";Cell ID=" << +m_cellId
       << ";CQI=" << +m_cqi
       << ";Component Carrier ID=" << +m_componentCarrierId
       << ")";

    return ss.str();
}

OranReportLteUeCqi::GetRnti() const
{
  NS_LOG_FUNCTION(this);
  return m_rnti;
}

OranReportLteUeCqi::GetCellId() const
{

  return m_cellId;
}

uint8_t
OranReportLteUeCqi::GetCqi() const
{
  NS_LOG_FUNCTION(this);
  return m_cqi;
}

uint16_t
OranReportLteUeCqi::GetComponentCarrierId() const
{
  NS_LOG_FUNCTION(this);
  return m_componentCarrierId;
}
